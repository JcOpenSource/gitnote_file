<!DOCTYPE html>

<html>

<head>

<meta charset="utf-8">

<!-- 引入 ECharts 文件 -->

<script src="echarts.js"></script>

<!-- 引入 vintage 主题 -->

<script src="infographic.js"></script>

</head>

<body>

<!-- 1.为 ECharts 准备一个具备大小（宽高）的 DOM -->

<div id="main" style="width: 600px;height:400px;"></div>

<!-- id名字 style样式 -->


<script type="text/javascript">

// 基于准备好的dom，初始化echarts实例

// var myChart = echarts.init(document.getElementById('main'));

// 2.首先拿到dom的节点

var domNode = document.getElementById('main');

// 3.然后初始化

var myChart = echarts.init(domNode, 'infographic');


// 4.指定图表的配置项和数据

var option = {

title : {

link: 'http:www.baidu.com',

text: 'xxxxx',

subtext: 'aaaaa',

x:'center'

},

tooltip : {

trigger: 'item',

formatter: "{a} <br/>{b} : {c} ({d}%)"

},

legend: {

orient: 'vertical',

left: 'left',

data: ['直接访问','邮件营销','联盟广告','视频广告','搜索引擎']

},

series : [

{

name: '访问来源',

type: 'pie',

radius : '55%',

center: ['50%', '60%'],

data:[

{value:335, name:'直接访问'},

{value:310, name:'邮件营销'},

{value:234, name:'联盟广告'},

{value:135, name:'视频广告'},

{value:1548, name:'搜索引擎'}

],

itemStyle: {

emphasis: {

shadowBlur: 10,

shadowOffsetX: 0,

shadowColor: 'rgba(0, 0, 0, 0.5)'

}

}

}

]

};


// 5.使用刚指定的配置项和数据显示图表。

myChart.setOption(option);

</script>

</body>

</html>
